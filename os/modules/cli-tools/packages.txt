# zsh — default interactive shell for sideral. Switch via `ujust chsh zsh`
# (or bash/nu for alternatives). Sideral's /etc/zshrc +
# /etc/zsh/sideral-cli-init.zsh wire it up with all CLI tools.
zsh
# zsh-syntax-highlighting + zsh-autosuggestions — bring zsh to
# parity with fish's two killer interactive features (red-on-invalid
# command coloring + greyed-out autosuggestions from history). Both
# from Fedora main, source-loaded by sideral-cli-init.zsh — no plugin
# manager needed for two source lines. Users who want bigger plugin
# stacks (oh-my-zsh, prezto, zinit) install at user level on top.
zsh-syntax-highlighting
zsh-autosuggestions
# rclone — CLI cloud-storage frontend (Google Drive, S3, B2, Dropbox,
# OneDrive, etc.). Powers `ujust gdrive-mount` for FUSE-mounting a
# Google Drive remote at ~/gdrive. Also useful for one-shot copies
# (`rclone copy ~/Documents gdrive:Backup/`) without a persistent mount.
rclone
# fuse3 — required for `rclone mount`. Usually pulled transitively by
# rclone but listing explicitly so removing rclone via override doesn't
# leave the mount recipe broken on systems that don't already have it.
fuse3
# chromium — second browser, useful for headless automation (puppeteer/
# playwright), web-app debugging, and as a fallback when Zen Browser
# can't render a particular site. Installed but HIDDEN from the GNOME
# app grid via hide-chromium.sh in this same module — the .desktop
# file gets `NoDisplay=true` patched in. Invokable from the shell as
# `chromium-browser`.
chromium
# ghostty — GPU-accelerated terminal (Terra repo). Per-user config
# symlinked into ~/.config/ghostty/config by sideral-stow-defaults on
# first login.
ghostty
# zed — GPU-accelerated, collaborative GUI editor (Terra repo, stable
# channel). Set as $EDITOR and $VISUAL by sideral-stow-defaults so git
# commit, sudoedit, mise edit, etc. all open a Zed buffer and block
# until it closes (`zed --wait`). Replaces the previous helix/code split.
zed
